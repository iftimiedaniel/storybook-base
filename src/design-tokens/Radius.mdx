# Border Radius

Below is the list of CSS variables for `border-radius` using increments of 0.25.
These are defined based on [Base spacing rules](?path=/docs/design-tokens-spacing--docs#base-spacing-rules)

{borderRadius()}

export function borderRadius() {
    const colorTokens = parseCSSVariables(cssString);
    const categoriesToInclude = ["border"];
    const filteredColorTokens = filterCategories(colorTokens, categoriesToInclude);

  return (
    <table>
      <thead>
        <tr>
          <th width="30%">Variable</th>
          <th>Value</th>
          <th width="10%">Preview</th>
        </tr>
      </thead>
      <tbody>
        {Object.keys(filteredColorTokens).map((tokenName) => (
            <tr key={tokenName}>
              <td align="center">--{tokenName}</td>
              <td align="center">{filteredColorTokens[tokenName]}</td>
              <td align="center">
                <div className='sb-addon' style={{ borderRadius: filteredColorTokens[tokenName] }}></div>
            </td>
            </tr>
          ))}
      </tbody>
    </table>
  );
}
