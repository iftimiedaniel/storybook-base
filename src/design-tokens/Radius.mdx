# Border Radius

Below is the list of **CSS variables** for `border-radius` using increments of 0.25.
Base color variables can be found [here](?path=/docs/design-tokens-global--docs#spacing).

{borderRadius()}

[How to use CSS variables](?path=/docs/design-tokens-global--docs#how-to-use-css-variables)

export function borderRadius() {
    const colorTokens = parseCSSVariables(cssString);
    const categoriesToInclude = ["border"];
    const filteredColorTokens = filterCategories(colorTokens, categoriesToInclude);

  return (
    <div class="design-token-card">
      <table>
        <thead>
          <tr>
            <th align="left" width="60%">Variable Name</th>
            <th align="left">Value</th>
            <th width="1%">Preview</th>
          </tr>
        </thead>
        <tbody>
          {Object.keys(filteredColorTokens).map((tokenName) => (
              <tr key={tokenName}>
                <td align="left"><pre>--{tokenName}</pre></td>
                <td align="left"><input type="text" readOnly value={filteredColorTokens[tokenName]} /></td>
                <td align="center">
                  <div className='design-token-card__preview design-token-card__preview--highlighted' style={{ borderRadius: filteredColorTokens[tokenName] }}></div>
              </td>
              </tr>
            ))}
        </tbody>
      </table>
    </div>
  );
}
