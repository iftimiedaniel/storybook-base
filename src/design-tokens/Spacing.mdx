# Spacing 

Below is the list of **CSS variables** for spacing. Base spacing variables can be found [here](?path=/docs/design-tokens-global--docs#spacing).

{spaces()}

[How to use CSS variables](?path=/docs/design-tokens-global--docs#how-to-use-css-variables)

export function spaces() {
  const colorTokens = parseCSSVariables(cssString);
  const spacings = filterCategories(colorTokens, ["space"]);

  return (
    <div className="design-token-card">
      <table>
        <thead>
          <tr>
            <th align="left" width="60%">Variable Name</th>
            <th align="left">Value</th>
            <th width="1%">Preview</th>
          </tr>
        </thead>
        <tbody>
          {Object.entries(spacings).map(([tokenName, tokenValue]) => (
            <tr key={tokenName}>
              <td align="left"><pre>--{tokenName}</pre></td>
              <td align="left"><input type="text" readOnly value={tokenValue} /></td>
              <td align="center">
                  <div className="design-token-card__preview design-token-card__preview--empty" style={{ height: tokenValue }}></div>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}
