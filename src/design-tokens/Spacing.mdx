# Spacing

Below is the list of CSS variables for spacing using increments of 0.25.
Everything's measured in `rems` where `1rem = 16px`.

### Base spacing rules
{pts()}

### Global spacing rules
{spaces()}

export function pts() {
    const colorTokens = parseCSSVariables(cssSpacesString);
    const categoriesToInclude = ["8"];
    const tokensToExclude = ["8-pt-numbers-full"];
    const filteredColorTokens = filterCategories(colorTokens, categoriesToInclude, tokensToExclude);

  return (
    <div class="design-token-card">
      <table>
        <thead>
          <tr>
            <th align="left" width="50%">Variable Name</th>
            <th align="left">Value</th>
            <th width="15%">Preview</th>
          </tr>
        </thead>
        <tbody>
          {Object.keys(filteredColorTokens).map((tokenName) => (
              <tr key={tokenName}>
                <td align="left"><pre>--{tokenName}</pre></td>
                <td align="left">{filteredColorTokens[tokenName]}</td>
                <td align="center">
                  <div className='design-token-card__preview design-token-card__preview--empty' style={{ height: filteredColorTokens[tokenName] }}></div>
              </td>
              </tr>
            ))}
        </tbody>
      </table>
    </div>
  );
}

export function spaces() {
  const colorTokens = parseCSSVariables(cssString);
  const spacings = filterCategories(colorTokens, ["space"]);

  return (
    <div class="design-token-card">
      <table>
        <thead>
          <tr>
            <th align="left" width="50%">Variable Name</th>
            <th align="left">Value</th>
            <th width="15%">Preview</th>
          </tr>
        </thead>
        <tbody>
          {Object.entries(spacings).map(([tokenName, tokenValue]) => (
            <tr key={tokenName}>
              <td align="left"><pre>--{tokenName}</pre></td>
              <td align="left"><input type="text" readonly value={tokenValue} /></td>
              <td align="center">
                  <div className='design-token-card__preview design-token-card__preview--empty' style={{ height: tokenValue }}></div>
              </td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}
