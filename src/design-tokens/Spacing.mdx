# Spacing

Below is the list of CSS variables for spacing using increments of 0.25.
Everything's measured in `rems` where `1rem = 16px`.

### Base spacing rules
{pts()}

### Global spacing rules
{spaces()}

export function pts() {
    const colorTokens = parseCSSVariables(cssString);
    const categoriesToInclude = ["8"];
    const tokensToExclude = ["8-pt-numbers-full"];
    const filteredColorTokens = filterCategories(colorTokens, categoriesToInclude, tokensToExclude);

  return (
    <table>
      <thead>
        <tr>
          <th width="30%">Variable</th>
          <th>Value</th>
          <th width="10%">Preview</th>
        </tr>
      </thead>
      <tbody>
        {Object.keys(filteredColorTokens).map((tokenName) => (
            <tr key={tokenName}>
              <td align="center">--{tokenName}</td>
              <td align="center">{filteredColorTokens[tokenName]}</td>
              <td align="center">
                <div className='sb-addon' style={{ height: filteredColorTokens[tokenName] }}></div>
            </td>
            </tr>
          ))}
      </tbody>
    </table>
  );
}

export function spaces() {
  const colorTokens = parseCSSVariables(cssString);
  const spacings = filterCategories(colorTokens, ["space"]);

  return (
    <table>
      <thead>
        <tr>
          <th width="30%">Variable</th>
          <th>Value</th>
          <th width="10%">Preview</th>
        </tr>
      </thead>
      <tbody>
        {Object.entries(spacings).map(([tokenName, tokenValue]) => (
          <tr key={tokenName}>
            <td align="center">--{tokenName}</td>
            <td align="center">{tokenValue}</td>
            <td align="center">
                <div className='sb-addon' style={{ height: tokenValue }}></div>
            </td>
          </tr>
        ))}
      </tbody>
    </table>
  );
}
